// Focus first focusable element const focusable = modal.querySelectorAll('button, [href], input, select, textarea, [tabindex]:not([tabindex="-1"])'); if (focusable.length > 0) focusable[0].focus(); } }); }); // Close modal document.querySelectorAll('.close-modal').forEach(closeBtn => { closeBtn.addEventListener('click', function() { const modal = this.closest('.modal'); if (modal) { modal.classList.remove('show');
